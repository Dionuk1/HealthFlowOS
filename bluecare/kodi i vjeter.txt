        function saveBooking() {
            const doc = document.getElementById('bookDoc').value;
            const date = document.getElementById('bookDate').value;
            const time = document.getElementById('bookTime').value;

            if(!date || !time) return alert('Ju lutem zgjedhni datën dhe orën!');

            const newApp = { doc, date, time, status: 'E konfirmuar' };
            appointments.unshift(newApp);

            alert(`Sukses! Vizita me ${doc} u rezervua për ${date} në ora ${time}.`);
            toggleModal('bookModal');
            
            if(!document.getElementById('app-shell').classList.contains('hidden')) {
                renderDashboard('user');
            }
        }

        function sendChat() {
            const input = document.getElementById('chatInput');
            const msgs = document.getElementById('chat-msgs');
            if(!input.value) return;

            const uMsg = document.createElement('div');
            uMsg.className = 'msg user animate__animated animate__fadeInRight';
            uMsg.innerText = input.value;
            msgs.appendChild(uMsg);

            const text = input.value.toLowerCase();
            input.value = '';
            msgs.scrollTop = msgs.scrollHeight;

            setTimeout(() => {
                const bMsg = document.createElement('div');
                bMsg.className = 'msg bot animate__animated animate__fadeInLeft';
                if(text.includes('dhimbje')) bMsg.innerText = "Më vjen keq që po ndjeheni ashtu. AI sugjeron të pushoni dhe të pini ujë. Nëse vazhdon, rezervoni një vizitë te Dr. Elena.";
                else if(text.includes('kush')) bMsg.innerText = "Unë jam HealthFlow AI, asistenti juaj mjekësor 24/7.";
                else bMsg.innerText = "Mund t'ju ndihmoj të gjeni një doktor ose të kontrolloni orarin tuaj.";
                msgs.appendChild(bMsg);
                msgs.scrollTop = msgs.scrollHeight;
            }, 800);
        }
